Consider the following tables.
Student
| StudentID | Name | CourseID |
| --------- | ----- | -------- |
| 1 | Rahul | 201 |
| 2 | Neha | 202 |
| 3 | Aman | NULL |
Course
| CourseID | CourseName |
| -------- | ---------- |
| 201 | Python |
| 202 | SQL |
Write an SQL query to display all students along with their course names
using a LEFT JOIN.

code

CREATE DATABASE College;
USE College;
CREATE TABLE Student (
    StudentID INT PRIMARY KEY,
    Name VARCHAR(50),
    CourseID INT
);
INSERT INTO Student (StudentID, Name, CourseID)
VALUES
(1, 'Rahul', 201),
(2, 'Neha', 202),
(3, 'Aman', NULL);
CREATE TABLE Course (
    CourseID INT PRIMARY KEY,
    CourseName VARCHAR(50)
);
INSERT INTO Course (CourseID, CourseName)
VALUES
(201, 'Python'),
(202, 'SQL');
SELECT * FROM Student;
SELECT * FROM Course;
SELECT
    Student.StudentID,
    Student.Name,
    Course.CourseName
FROM Student
LEFT JOIN Course
ON Student.CourseID = Course.CourseID;
